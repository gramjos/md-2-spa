Feature creation based on columns value

sql
SELECT
    Column1,
    Column2,
    CASE 
        WHEN Column1 IS NULL AND Column2 IS NOT NULL THEN Column2
        WHEN Column2 IS NULL AND Column1 IS NOT NULL THEN Column1
        WHEN Column1 IS NOT NULL AND Column2 IS NOT NULL AND Column1 <> Column2 THEN Column1 + ', ' + Column2
        ELSE Column1 -- or Column2, since they are the same if both are not null and equal
    END AS NewColumn
FROM
    YourTableName;
