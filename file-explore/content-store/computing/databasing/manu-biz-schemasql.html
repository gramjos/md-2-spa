sql
-- Create Employees table
CREATE TABLE Employees (
    EmployeeID SERIAL PRIMARY KEY,
    FirstName VARCHAR(255),
    LastName VARCHAR(255),
    Department VARCHAR(255),
    Position VARCHAR(255)
);

-- Create Components table
CREATE TABLE Components (
    ComponentID SERIAL PRIMARY KEY,
    Name VARCHAR(255),
    Description TEXT
);

-- Create Inventory table
CREATE TABLE Inventory (
    InventoryID SERIAL PRIMARY KEY,
    ComponentID INTEGER,
    Quantity INTEGER,
    CONSTRAINT fk_Component
        FOREIGN KEY (ComponentID)
        REFERENCES Components (ComponentID)
        ON DELETE CASCADE
);

-- Create ProductLines table
CREATE TABLE ProductLines (
    ProductID SERIAL PRIMARY KEY,
    ProductName VARCHAR(255),
    Description TEXT
);

-- Create ProductComponents table (junction table for many-to-many relationship)
CREATE TABLE ProductComponents (
    ProductID INTEGER,
    ComponentID INTEGER,
    Quantity INTEGER,
    PRIMARY KEY (ProductID, ComponentID),
    CONSTRAINT fk_Product
        FOREIGN KEY (ProductID)
        REFERENCES ProductLines (ProductID)
        ON DELETE CASCADE,
    CONSTRAINT fk_Component
        FOREIGN KEY (ComponentID)
        REFERENCES Components (ComponentID)
        ON DELETE CASCADE
);

-- Create SalesOrder table
CREATE TABLE SalesOrder (
    OrderID SERIAL PRIMARY KEY,
    ProductID INTEGER,
    EmployeeID INTEGER,
    OrderDate DATE,
    Quantity INTEGER,
    CONSTRAINT fk_Product
        FOREIGN KEY (ProductID)
        REFERENCES ProductLines (ProductID)
        ON DELETE SET NULL,
    CONSTRAINT fk_Employee
        FOREIGN KEY (EmployeeID)
        REFERENCES Employees (EmployeeID)
        ON DELETE SET NULL
)
