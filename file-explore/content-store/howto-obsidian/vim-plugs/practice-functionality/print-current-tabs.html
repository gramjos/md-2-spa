In this contrived project I will use: my .obsidian.vimrc to call and external javascript file that accesses the Obsidian API.

In the .obsidian.vimrc

vim
exmap ShowTabNames jsfile .vim_tricks/ShowTabNames.js {showMe()}
map <Space>t :ShowTabNames

In ShowTabNames.js

js
function showMe() {
    // Get all leaves (tabs) in the workspace
    const leaves = app.workspace.getLeavesOfType('markdown');

    // Collect the names of the open tabs
    const tabNames = leaves.map(leaf => {
        if (leaf.view.file) {
            return leaf.view.file.name;
        } else {
            return 'Untitled';
        }
    });

    // Join the tab names into a single string
    const tabNamesString = tabNames.join('\n');

    // Insert the tab names at the cursor position
    const editor = app.workspace.activeLeaf.view.editor;
    const cursor = editor.getCursor();
    editor.replaceRange(tabNamesString, cursor);

    // Also print the tab names to the console
    console.log('Open Tabs:\n' + tabNamesString);
}
